def sum(a, b):
    """Sum of two floats.

    Examples:
        >>> sum(3.0, 4.0)
        7.0

    Args:
        a (float): First argument
        b (float): Second argument

    Returns:
        float: Returns the sum operation of `a` and `b`.

    """
    return a + b
